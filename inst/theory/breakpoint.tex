\section{Subject specific, single breakpoint, single slope}
\label{sec:org0978280}

\subsection{Direct optimisation}
\label{sec:orge803f2f}

Consider a response variable \(Y\) and an explanatory variable \(X\)
related by:
\begin{align*}
Y = \beta X - \beta (X - \psi)_+ + \varepsilon
\end{align*}

\noindent where \((\beta,\psi) \in \Real^2\), \(\varepsilon
\sim \Gaus[0,\sigma^2]\), and \((x)_+=x\) if \(x>0\) and 0
otherwise. Introduce \(\Theta = (\beta,\psi,\sigma^2)\), we can
express the likelihood relative to \(n\) iid observations as:
\begin{align*}
\Likelihood(\Theta) = \prod_{i=1}^n \frac{1}{\sqrt{2\pi\sigma^2}}\exp\left(-\frac{(Y_i - \beta X_i + \beta (X_i - \psi)_+)^2}{2\sigma^2}\right)
\end{align*}
and the log-likelihood as:
\begin{align*}
\likelihood(\Theta) = - \frac{n}{2} \log(2\pi) - \frac{n}{2} \log(\sigma^2) - \sum_{i=1}^n \frac{\left(Y_i - \beta X_i + \beta (X_i - \psi)_+\right)^2}{2\sigma^2}
\end{align*}

\noindent Maximizing the likelihood with respect to \(\Theta_\mu =
(\beta,\psi)\) is equivalent to minimizing the residual sum of
squared (RSS):
\begin{align*}
RSS(\Theta_\mu) = \sum_{i=1}^n (Y_i - \beta X_i + \beta (X_i - \psi)_+)^2
\end{align*}

One approach to estimate the model parameters is to directly minimize
this log-likelihood using, for instance grid search. This may be
numerically inefficient but does not require any smoothness assumption
on the likelihood.
\begin{itemize}
\item this is what \texttt{optim.lmbreak\_NLMINB} in the lmbreak package does (set
argument \texttt{control} to \texttt{list(optimizer = "nlminb")}). \texttt{nlminb} seems
to employ a quasi Newton method though, some does require some
smoothness.
\end{itemize}

\subsection{Gradient descent}
\label{sec:org8d305e5}

One can re-write the residual sum of squares as:
\begin{align*}
RSS(\Theta_\mu) = \sum_{i=1}^n \varepsilon_i(\psi)^2  = Y (I-H(\psi))\trans{(I-H(\psi))}\trans{Y} = Y (I-H(\psi))\trans{Y}
\end{align*}
where \(H(\psi) =
X(\psi)\left(\trans{X}(\psi)X(\psi)\right)^{-1}\trans{X}(\psi)\) is
the hat matrix. It is idempotent and so is \(I-H\) where \(I\) denote
the identity matrix. Therefore a possible derivative w.r.t. \(\psi\) of the RSS is:
\begin{align*}
\frac{\partial RSS(\Theta_\mu)}{\partial \psi} =& - Y \frac{\partial H(\psi)}{\partial \psi} \trans{Y} \\
 =& - Y \frac{\partial X(\psi)}{\partial \psi} \left(\trans{X}(\psi) X(\psi)\right)^{-1}\trans{X}(\psi) \trans{Y} \\
  & + Y X(\psi) \left(\trans{X}(\psi) X(\psi)\right)^{-1} \left(\frac{\partial \trans{X}(\psi)}{\partial \psi}X(\psi)+\trans{X}(\psi)\frac{\partial X(\psi)}{\partial \psi} \right) \left(\trans{X}(\psi) X(\psi)\right)^{-1}\trans{X}(\psi) \trans{Y} \\
  &  - Y X(\psi) \left(\trans{X}(\psi)  X(\psi)\right)^{-1} \frac{\partial \trans{X}(\psi)}{\partial \psi} \trans{Y}
\end{align*}
The first and last term being scalar and transpose of each other they
are equal. Similarly the two middle terms are also scalar and
transpose of each other so equal. Therefore:
\begin{align*}
\frac{\partial RSS(\Theta_\mu)}{\partial \psi}
=& - 2 Y \frac{\partial X(\psi)}{\partial \psi} \left(\trans{X}(\psi) X(\psi)\right)^{-1}\trans{X}(\psi) \trans{Y} \\
 & + 2 Y X(\psi) \left(\trans{X}(\psi) X(\psi)\right)^{-1} \left(\trans{X}(\psi)\frac{\partial X(\psi)}{\partial \psi} \right) \left(\trans{X}(\psi) X(\psi)\right)^{-1}\trans{X}(\psi) \trans{Y} \\
=& 2 Y \left(- \frac{\partial X(\psi)}{\partial \psi} + X(\psi) \left(\trans{X}(\psi) X(\psi)\right)^{-1} \trans{X}(\psi)\frac{\partial X(\psi)}{\partial \psi} \right) \left(\trans{X}(\psi) X(\psi)\right)^{-1}\trans{X}(\psi) \trans{Y} \\
=& 2 Y \left(X(\psi) \left(\trans{X}(\psi) X(\psi)\right)^{-1} \trans{X}(\psi) - I \right) \frac{\partial X(\psi)}{\partial \psi} \left(\trans{X}(\psi) X(\psi)\right)^{-1}\trans{X}(\psi) \trans{Y}
\end{align*}
So it only remains to evaluate the (sub-)derivatives of \(X(\psi)\):
\begin{align*}
\frac{\partial X(\psi)}{\partial \psi} = \frac{\partial (X-\psi)\Ind[X>\psi]}{\partial \psi}
 = -\Ind[X>\psi] + \Ind[X=\psi]\delta
\end{align*}
where \(\delta \in [0,1]\).

\bigskip

So a second approach to estimate the model parameters is to perform a
gradient descent based on sub-derivatives
\begin{itemize}
\item this is what \texttt{optim.lmbreak\_BFGS} in the lmbreak package does (set
argument \texttt{control} to \texttt{list(optimizer = "BFGS")}). \(\delta\) is then randomly
sampled in a uniform distribution whose support is \([0,1]\).
\end{itemize}

\clearpage


\noindent \textbf{Note}: another approach to 'profile-out' the \(\beta\), is to first
 plug-in the OLS estimator:
\begin{align*}
\tilde{\beta}(\psi) &= \frac{\sum_{i=1}^n X_i Y_i - (X_i - \psi)_+ Y_i}{\sum_{i=1}^n (X_i + (X_i - \psi)_+)^2} = \frac{\sum_{i=1}^n \Ind[X_i \leq \psi] X_i Y_i}{\sum_{i=1}^n \Ind[X_i \leq \psi] X^2_i} 
\end{align*}

and then minimize w.r.t. \(\psi\):
\begin{align*}
\likelihood(\psi) &= \sum_{i=1}^n (Y_i - \tilde{\beta}(\psi) X_i + \tilde{\beta}(\psi) (X_i - \psi)_+)^2
\end{align*}

So the first 'derivative' should solve:
\begin{align*}
0 =& -2 \sum_{i=1}^n \left[
  \frac{\partial \tilde{\beta}(\psi)}{\partial \psi} (X_i - (X_i - \psi)_+)
- \tilde{\beta}(\psi)\frac{\partial (X_i - \psi)_+}{\partial \psi}\right]
(Y_i - \tilde{\beta}(\psi) X_i + \tilde{\beta}(\psi)(X_i - \psi)_+ ) \\
0 =& \frac{\partial \tilde{\beta}(\psi)}{\partial \psi} \sum_{i=1}^n (X_i - (X_i - \psi)_+)(Y_i - \tilde{\beta}(\psi) X_i + \tilde{\beta}(\psi)(X_i - \psi)_+ ) \\
& - \tilde{\beta}(\psi)\sum_{i=1}^n\frac{\partial (X_i - \psi)_+}{\partial \psi}(Y_i - \tilde{\beta}(\psi) X_i + \tilde{\beta}(\psi)(X_i - \psi)_+ ) 
\end{align*}
Noticing that \(\tilde{\beta}(\psi)\) is a step function with jumps at
\((X_i)_{i\in\{1,\ldots,n\}}\). If the breakpoint does not coincide
with any datapoint, it has derivative 0 w.r.t. \(\psi\). Under such
assumption, the previous expression simplifies into:

\begin{align*}
0 =& - \tilde{\beta}(\psi)\sum_{i=1}^n \Ind[X_i \geq \psi] (Y_i - \tilde{\beta}(\psi) \psi)  \\
\frac{1}{n}\sum_{i=1}^n \Ind[X_i \geq \psi] Y_i =& \tilde{\beta}(\psi) \psi
\end{align*}

So the breakpoint should be such that the average post-breakpoint value equal the fitted plateau.

\subsection{Model}
\label{sec:orge53288e}

Consider a set of \(n\) subjects each undergoing \(T\) measurements at
equally spaced timepoints \(1,\ldots,T\) of an outcome variable
\(Y\). We consider the following random slope and random breakpoint
model:
\begin{align*}
Y_{i,t} =& \beta_i t - \beta_i (t - \psi_i)_+ + \varepsilon_{i,t}
\end{align*}
where \(\beta_i = \beta + u_i\) and \(\psi_i = \psi + v_i\) and:
\begin{align*}
\begin{bmatrix}
u_i    \\         
v_i    \\
\varepsilon_{i,t} \\
\end{bmatrix}
= \Gaus \left(
\begin{bmatrix}
0    \\         
0    \\
0    \\
\end{bmatrix},
\begin{bmatrix}
\tau_{u,u} & \tau_{u,v} & 0 \\         
\tau_{u,v} & \tau_{v,v} & 0 \\         
0          & 0          & \sigma^2 \\         
\end{bmatrix}
\right)
\end{align*}
We further denote \(\Sigma_\tau= \begin{bmatrix} \tau_{u,u} &
\tau_{u,v} \\ \tau_{u,v} & \tau_{v,v} \end{bmatrix}\) whose inverse is
\(\Sigma^{-1}_\tau = \begin{bmatrix} w_{u,u} & w_{u,v} \\ w_{u,v} &
w_{v,v} \end{bmatrix} =
\frac{1}{\tau_{u,u}\tau_{v,v}-\tau^2_{u,v}} \begin{bmatrix} \tau_{v,v}
& -\tau_{u,v} \\ -\tau_{u,v} & \tau_{u,u} \end{bmatrix}\). We now can re-write the model as:
\begin{align*}
Y_{i,t} =& \beta (t - (t - \psi - v_i)_+) + u_i (t - (t - \psi - v_i)_+) + \varepsilon_{i,t} \\
        =& \mu(t,\beta,\psi,u_i,v_i) + \varepsilon_{i,t}
\end{align*}
where \(\mu(t,\beta,\psi,u_i,v_i) = \beta t + u_i t\) when \(t \leq
\psi + v_i\) \newline and \(\mu(t,\beta,\psi,u_i,v_i) = \beta (\psi +
v_i) + u_i (\psi + v_i)\) otherwise.
\subsection{Likelihood}
\label{sec:orgba80f7c}
\subsubsection{Bivariate integral}
\label{sec:org4f8b82c}

Introduce \(\Theta =
(\beta,\psi,\tau_{u,u},\tau_{u,v},\tau_{v,v},\sigma^2)\), we can
express the likelihood relative to \(n\) iid subjects as:
\begin{align*}
\Likelihood(\Theta) &= \prod_{i=1}^n \prod_{t=1}^T \int_{u_i=-\infty}^{+\infty}\int_{v_i=-\infty}^{+\infty} \frac{1}{\sqrt{2\pi\sigma^2}}\exp\left(-\frac{(Y_i - \mu(t,\beta,\psi,u_i,v_i))^2}{2\sigma^2}\right)f_{\tau}(u,v) du_i dv_i \\
\text{ where } f_{\tau}(u,v) =& \frac{1}{2\pi \left|\Sigma_\tau\right|^{1/2}} \exp\left(-\frac{1}{2}\begin{bmatrix}u & v \end{bmatrix}\Sigma^{-1}_\tau\begin{bmatrix}u\\ v \end{bmatrix}\right) \\
              =& \frac{1}{2\pi \sqrt{\tau_{u,u}\tau_{v,v}-\tau^2_{u,v}}} \exp\left(-\frac{u^2 w_{u,u}+2uv w_{u,v}+v^2 w_{v,v}}{2}\right)
\end{align*}

\subsubsection{Nested univariate integrals}
\label{sec:orgf9514e2}


To simplify the numerical integration for a bivariate integral to
univariate integral, we start by splitting second integral depending
on whether \(t \leq \psi + v_i\) i.e. \(v_i \geq t - \psi\):
\begin{align*}
\Likelihood(\Theta) =& \prod_{i=1}^n \prod_{t=1}^T \int_{u_i=-\infty}^{+\infty}\int_{v_i=t-\psi}^{+\infty} \frac{1}{\sqrt{2\pi\sigma^2}}\exp\left(-\frac{(Y_i - \beta t - u_i t)^2}{2\sigma^2}\right)f_{\tau}(u_i,v_i) du_i dv_i \\
                   & + \prod_{i=1}^n \prod_{t=1}^T \int_{u_i=-\infty}^{+\infty}\int_{v_i=-\infty}^{t-\psi} \frac{1}{\sqrt{2\pi\sigma^2}}\exp\left(-\frac{(Y_i - \beta (\psi + v_i) - u_i (\psi + v_i))^2}{2\sigma^2}\right)f_{\tau}(u_i,v_i) du_i dv_i \\
                   =& \frac{1}{(2\pi)^{3nT/2}\sigma^{nT}(\tau_{u,u}\tau_{v,v}-\tau^2_{u,v})^{nT/2}} \prod_{i=1}^n \prod_{t=1}^T \left(\Likelihood_{1,i,t}(\Theta) + \Likelihood_{2,i,t}(\Theta) \right)
\end{align*}
where we introduce the notation:
\begin{align*}
\Likelihood_{1,i,t}(\Theta) =& \int_{u_i=-\infty}^{+\infty}\int_{v_i=t-\psi}^{+\infty} \exp\left(-\frac{(Y_i - \beta t - u_i t)^2}{2\sigma^2} - \frac{u_i^2 w_{u,u}+2u_i v_i w_{u,v}+v_i^2 w_{v,v}}{2}\right) du_i dv_i \\
\Likelihood_{2,i,t}(\Theta) =& \int_{u_i=-\infty}^{+\infty}\int_{v_i=-\infty}^{t-\psi} \exp\left(-\frac{(Y_i - \beta (\psi + v_i) - u_i (\psi + v_i))^2}{2\sigma^2} - \frac{u_i^2 w_{u,u}+2u_i v_i w_{u,v}+v_i^2 w_{v,v}}{2}\right) du_i dv_i
\end{align*}
Using that:
\begin{align*}
\int_{a}^{b} \exp\left( - \frac{1}{2}(A v_i^2 + 2 B v_i) \right) dv_i 
&= \int_{a}^{b} \exp\left( - \frac{A}{2}(v_i^2 + 2 \frac{B}{A} v_i) \right) dv_i  \\
&= \int_{a}^{b} \exp\left( - \frac{A}{2}(v_i + \frac{B}{A})^2 + \frac{B^2}{2 A} \right) dv_i  \\ 
&= \sqrt{\frac{2 \pi}{A}}\exp\left(\frac{B^2}{2 A}\right)\frac{1}{\sqrt{2 \pi A^{-1}}} \int_{a}^{b} \exp\left( - \frac{(v^2_i- -\frac{B}{A})^2}{2A^{-1}} \right) dv_i \\
&= \sqrt{\frac{2 \pi}{A}}\exp\left(\frac{B^2}{2 A}\right) \left(\Phi_{\mu=-\frac{B}{A},\sigma^2=A^{-1}}(b) - \Phi_{\mu=-\frac{B}{A},\sigma^2=A^{-1}}(a) \right) \\
&= \sqrt{\frac{2 \pi}{A}}\exp\left(\frac{B^2}{2 A}\right) \left(\Phi\left(\frac{bA+B}{\sqrt{A}}\right) - \Phi\left(\frac{aA+B}{\sqrt{A}}\right) \right)
\end{align*}
where \(\Phi_{\mu,\sigma^2}(\bullet)= \Phi(\frac{\bullet-\mu}{\sigma})\) denotes
the Cumulative Distribution Function (CDF) of a normal distribution
with mean \(\mu\) and variance \(\sigma^2\). For the first term, using
\(a=t-\psi\), \(b=+\infty\), \(A=w_{v,v}\), and \(B = u_i w_{u,v}\),
we obtain:
\begin{align*}
\Likelihood_{1,i,t}(\Theta) =&
\sqrt{\frac{2\pi}{w_{v,v}}} \int_{-\infty}^{+\infty} \exp\left(-\frac{(Y_i - \beta t - u_i t)^2}{2\sigma^2} - \frac{u_i^2 (w_{u,u} -  \frac{w^2_{u,v}}{w_{v,v}})}{2} \right)
                                             \left(1 - \Phi\left(\frac{w_{v,v} (t-\psi) + u_i w_{u,v}}{\sqrt{w_{v,v}}}\right)\right)
\end{align*}
which can be evaluated using numerical intergration
(e.g. \texttt{stats::integrate} and \texttt{stats::pnorm} - see function next
page). A similar approach can be used to evaluate the second term.
\begin{adjustwidth}{-7em}{-7em}
\begin{align*}
&\Likelihood_{2,i,t}(\Theta) = \int_{u_i=-\infty}^{+\infty}\int_{v_i=-\infty}^{t-\psi} \exp\left(-\frac{(Y_i - \psi (\beta + u_i) - v_i (\beta + u_i))^2}{2\sigma^2} - \frac{u_i^2 w_{u,u}+2u_i v_i w_{u,v}+v_i^2 w_{v,v}}{2}\right) du_i dv_i \\
=& \int_{u_i=-\infty}^{+\infty}\int_{v_i=-\infty}^{t-\psi} \exp\left(-\frac{(Y_i - \psi (\beta + u_i))^2 + v^2_i (\beta + u_i)^2 - 2 v_i (\beta + u_i) (Y_i - \psi (\beta + u_i))}{2\sigma^2} - \frac{u_i^2 w_{u,u}+2u_i v_i w_{u,v}+v_i^2 w_{v,v}}{2}\right) du_i dv_i
\end{align*}
\end{adjustwidth}
Using \(a=-\infty\), \(b=t-\psi\), \(A=w_{v,v}+\frac{(\beta+u_i)^2}{\sigma^2}\), and
\(B = u_i w_{u,v} - h(u_i)\) where
\(h(u_i)=\frac{(\beta+u_i)(Y_i-\psi(\beta+u_i))}{\sigma^2}\), we obtain for \(\Likelihood_{2,i,t}(\Theta)\):

\begin{adjustwidth}{-7em}{-7em}
\begin{align*}
\sqrt{2\pi} \int_{-\infty}^{+\infty} \exp\left(-\frac{(Y_i - \psi (\beta + u_i))^2}{2\sigma^2} - \frac{u_i^2 w_{u,u} -  \frac{(u_i w_{u,v}-h(u_i))^2}{w_{v,v}+(\beta+u_i)^2}}{2} \right)
                                             \frac{\Phi\left( \frac{(w_{v,v}+(\beta+u_i)^2) (t-\psi) + u_i w_{u,v} - h(u_i)}{\sqrt{w_{v,v}+(\beta+u_i)^2}}\right)}{\sqrt{w_{v,v}+(\beta+u_i)^2}} du_i
\end{align*}
\end{adjustwidth}

\subsection{Direct optimisation}
\label{sec:orga57bf3b}

\noindent We can now search for the model parameters maximizing the likelihood:
\begin{itemize}
\item using a transformation to constrain the variance-covariance matrix
of the random effect to be positive
\end{itemize}
\begin{lstlisting}[language=r,numbers=none]
res.optim0 <- optim(
  ## initialize at the DGM value
  par = c(tau_uu.log = log(MtauT[1,1]),
          rho_uv = atanh(cov2cor(MtauT)[1,2]),
          tau_vv.log = log(MtauT[2,2])),
  fn = function(theta){
    theta["tau_uu"] <- exp(theta["tau_uu.log"])
    theta["tau_vv"] <- exp(theta["tau_vv.log"])
    theta["tau_uv"] <- sqrt(theta["tau_uu"]*theta["tau_vv"])*tanh(theta["rho_uv"])
    iOut <- logLik_mixed10(Y = df.sim$Y, t = df.sim$X,
                           beta = betaT, psi = psiT,
                           tau_uu = theta["tau_uu"],
                           tau_uv = theta["tau_uv"],
                           tau_vv = theta["tau_vv"],
                           sigma2 = sigma2T,
                           indiv = FALSE)
    return(-iOut) ## optim is minimizing
  }, control = list(reltol = 1e-3))

\subsection{Profile likelihood}
\label{sec:org9e71292}

To simplify the optimisation problem we will find the optimal
expression of \((\beta,\psi,\sigma^2)\) given
\((\tau_{u,u},\tau_{u,v},\tau{v,v})\) and inject these in the
likelihood.

\bigskip

\noindent To do so we first derive the joint distribution of \((Y,u,v)\):
\begin{align*}
\Esp[Y] =& \beta t + \beta \Esp\left[(t-\psi-v_i)\Ind[t-\psi-v_i>0]\right]  + \Esp\left[u_i(t-\psi-v_i)\Ind[t-\psi-v_i>0]\right] \\
 =& \beta t + \beta (t-\psi) - \Esp\left[v_i\Ind[t-\psi-v_i>0]\right] + (t-\psi)\Esp\left[u_i\Ind[t-\psi-v_i>0]\right] - \Esp\left[u_i v_i\Ind[t-\psi-v_i>0]\right] \\
 =& \beta t + \beta (t-\psi) - \Esp\left[v_i\Ind[v_i < t-\psi]\right] + (t-\psi)\Esp\left[u_i\Ind[v_i<t-\psi]\right] - \Esp\left[u_i v_i\Ind[v_i<t-\psi]\right] \\
\end{align*}

We now focus on the expectation terms:
\begin{align*}
\Esp\left[v_i\Ind[v_i < t-\psi]\right] = \int_{-\infty}^{t-\psi} \frac{v_i}{\sqrt{2\pi \tau_{v,v}}} \exp\left(-\frac{v_i^2}{2\tau_{v,v}}\right)
\end{align*}
